let dragging = null; function startDrag(event, hand) { dragging = hand; event.preventDefault(); // Prevent default drag behavior document.addEventListener('mousemove', moveHand); document.addEventListener('mouseup', stopDrag); document.addEventListener('touchmove', moveHand, { passive: false }); document.addEventListener('touchend', stopDrag); document.addEventListener('touchcancel', stopDrag); // Handle touch cancel } function moveHand(event) { const { clientX, clientY } = event.touches ? event.touches[0] : event; const rect = clock.getBoundingClientRect(); const cx = rect.left + rect.width / 2; const cy = rect.top + rect.height / 2; const angle = Math.atan2(clientY - cy, clientX - cx) * (180 / Math.PI) + 90; if (dragging === 'startHour') { const hour = Math.round(angle / 30) % 12 || 12; sH.value = hour; } else if (dragging === 'startMinute') { const minute = Math.round(angle / 6) % 60; sM.value = minute; } else if (dragging === 'endHour') { const hour = Math.round(angle / 30) % 12 || 12; eH.value = hour; } else if (dragging === 'endMinute') { const minute = Math.round(angle / 6) % 60; eM.value = minute; } // Update both clocks updateClock(); updateEndClock(); } function stopDrag() { dragging = null; document.removeEventListener('mousemove', moveHand); document.removeEventListener('mouseup', stopDrag); document.removeEventListener('touchmove', moveHand); document.removeEventListener('touchend', stopDrag); document.removeEventListener('touchcancel', stopDrag); // Ensure all types are covered } // Event listeners for dragging hourCircle.addEventListener('mousedown', (e) => startDrag(e, 'startHour')); minuteCircle.addEventListener('mousedown', (e) => startDrag(e, 'startMinute')); endHourCircle.addEventListener('mousedown', (e) => startDrag(e, 'endHour')); endMinuteCircle.addEventListener('mousedown', (e) => startDrag(e, 'endMinute')); hourCircle.addEventListener('touchstart', (e) => startDrag(e, 'startHour'), { passive: false }); minuteCircle.addEventListener('touchstart', (e) => startDrag(e, 'startMinute'), { passive: false }); endHourCircle.addEventListener('touchstart', (e) => startDrag(e, 'endHour'), { passive: false }); endMinuteCircle.addEventListener('touchstart', (e) => startDrag(e, 'endMinute'), { passive: false });
